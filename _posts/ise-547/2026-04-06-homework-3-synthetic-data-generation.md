from scipy.stats import pearsonr

# Set seed for reproducibility
np.random.seed(42)

def generate_synthetic_cardio_data(n_patients=40):
    # 1. Generate Cholesterol (Independent variable)
    chol = np.random.uniform(150, 350, n_patients)
    
    # 2. Model the Negative Relationship for thalch
    # Formula: thalch = baseline - (factor * chol) + noise
    # We want thalch between 90-200
    noise = np.random.normal(0, 10, n_patients)
    thalch = 250 - (0.4 * chol) + noise
    thalch = np.clip(thalch, 90, 200)

    # 3. Generate other numerical columns within realistic ranges
    trestbps = np.random.uniform(90, 180, n_patients)
    # Logical interaction: higher oldpeak more likely if exang is 1
    exang = np.random.choice([0, 1], n_patients, p=[0.6, 0.4])
    oldpeak = np.zeros(n_patients)
    for i in range(n_patients):
        base_peak = np.random.uniform(0, 3)
        oldpeak[i] = base_peak + (2.0 if exang[i] == 1 else 0)
    oldpeak = np.clip(oldpeak, 0.0, 6.0)

    # 4. Generate Categorical columns
    fbs = np.random.choice([0, 1], n_patients, p=[0.8, 0.2])
    restecg = np.random.choice([0, 1, 2], n_patients)
    slope = np.random.choice([0, 1, 2], n_patients)
